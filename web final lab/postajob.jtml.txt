<script>
function postJob() {
  const jobTitle = document.getElementById("job-title").value;
  const jobDescription = document.getElementById("job-description").value;
  const jobType = document.getElementById("job-type").value;
  const jobLocation = document.getElementById("job-location").value;
  const applicationDeadline = document.getElementById("application-deadline").value;

  const jobPosting = {
    title: jobTitle,
    description: jobDescription,
    type: jobType,
    location: jobLocation,
    deadline: applicationDeadline
  };

 function postJob() {
  // Code to get job posting object from form inputs

  const xhr = new XMLHttpRequest();
  xhr.open("POST", "/post-job");
  xhr.setRequestHeader("Content-Type", "application/json");
  xhr.onload = function() {
    if (xhr.status === 200) {
      console.log("Job posted successfully");
    } else {
      console.log("Job posting failed");
    }
  };
  xhr.send(JSON.stringify(jobPosting));
}

}
const form = document.getElementById("job-posting-form");
form.addEventListener("submit", function(event) {
  event.preventDefault();
  postJob();
});
